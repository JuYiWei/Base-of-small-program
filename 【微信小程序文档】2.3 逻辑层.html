2.3 逻辑层

一、简介

  • 由 javascript 编写，但是不支持原生 javascript 能力，只能使用提供的 API；
  • 逻辑层处理数据更新视图，并接受视图的事件反馈；

二、注册程序

1. App()

  • 小程序生命周期和全局参数；
属性 类型 说明
onLaunch Function 小程序初始化
onShow Function 小程序显示，启动或后台进入
onHide Function 小程序隐藏，进入后台
onError Function 小程序脚本错误或者 api 异常
其他 Any 任意添加函数或者 Object
  • onLaunch、onShow 参数
参数 类型 说明
path String 打开小程序的路径
query Object 打开小程序的 query
scene Number 场景值
shareTicket String shareTicket 转发相关
referrerInfo Object 由另一个小程序打开时存在
referrerInfo.appId String 来源 appId
referrerInfo.extraData Object 来源数据
  • getApp()
// 获取小程序实例，主要用于获取 app 全局参数()
var data = getApp().appGlobalData;
  • 示例
App({
  testData: "test",
  testFun: function(obj) {
    console.log(obj);
  },
  /**
   * 当小程序初始化完成时，会触发 onLaunch（全局只触发一次）
   */
  onLaunch: function (options) {
    this.testFun('测试 公共函数 和 公共数据' + this.testData);
    console.log(JSON.stringify(options));
  },

  /**
   * 当小程序启动，或从后台进入前台显示，会触发 onShow
   */
  onShow: function (options) {
    // 同 onLaunch options
  },

  /**
   * 当小程序从前台进入后台，会触发 onHide
   */
  onHide: function () {
    
  },

  /**
   * 当小程序发生脚本错误，或者 api 调用失败时，会触发 onError 并带上错误信息
   */
  onError: function (msg) {
    console.log(msg);
  },
  appGlobalData: '全局数据'
})

四、注册页面

1. Page()

  • 页面生命周期和全局参数；
属性 类型 说明
data Object 页面初始数据 必须是 JSON 格式，setData 方式重绘 UI
onLoad Function 页面加载，只触发一次
onReady Function 页面初次渲染完成，初次渲染完成时
onShow Function 页面显示
onHide Function 页面隐藏
onUnload Function 页面释放
onPullDownRefresh Function 下拉
onReachBottom Function 上拉触底
onPageScroll Function 页面滚动
onShareAppMessage Function 右上角转发
其他 Any 任意添加函数或者 Object，当前页面 this 可以访问
  • 示例
Page({

  /**
   * 页面的初始数据
   */
  data: {
    count: 1
  },

  add: function (e) {
    this.setData({
      count: this.data.count + 1
    })
    console.log(e);
  },

  /**
   * 生命周期函数--监听页面加载
   */
  onLoad: function (options) {
    console.log("onLoad");
    console.log(this.route);// 当前路径
  },

  /**
   * 生命周期函数--监听页面初次渲染完成
   */
  onReady: function () {
    console.log("onReady");
  },

  /**
   * 生命周期函数--监听页面显示
   */
  onShow: function () {
    console.log("onShow");
  },

  /**
   * 生命周期函数--监听页面隐藏
   */
  onHide: function () {
    console.log("onHide");
  },

  /**
   * 生命周期函数--监听页面卸载
   */
  onUnload: function () {
    console.log("onUnload");
  },

  /**
   * 页面相关事件处理函数--监听用户下拉动作
   */
  onPullDownRefresh: function () {
    console.log("onPullDownRefresh");
  },

  /**
   * 页面上拉触底事件的处理函数
   */
  onReachBottom: function () {
    console.log("onReachBottom");
  },

  /**
   * 用户点击右上角分享
   */
  onShareAppMessage: function () {
    return {
      title: '转发标题',
      path: '/pages/home/index'
    }
  },

  /**
   * 用户滑动页面事件
   */
  onPageScroll: function (option) {
    console.log(option);
  }
})

六、模块化

1. 文件作用域

  • 变量与函数仅当前文件内有效
  • 全局实例需要在 App() 中设置

2. 模块化

  • 公共 js 文件作为模块，只能通过 module.exports 对外暴露接口（不建议直接使用 exports）

  • common.js

function say(name) {
  console.log('hello' + name);
}

module.exports.sayHello = say;
  • page.js
var common = require('../../common/common.js');// 引入文件，相对路径
common.sayHello('张三');
